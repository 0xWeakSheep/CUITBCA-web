## 概述

多页面动效设计方案旨在为 cuitbca.online 网站的不同页面类型创建统一而富有特色的视觉体验。每种页面类型都有其独特的用户目标和内容特点，因此需要量身定制的动效策略。本方案确保在保持品牌一致性的同时，为每个页面提供最适合其功能和内容的交互体验。

### 设计理念

**统一性与差异性并重**

- 核心动效元素在所有页面保持一致（如导航、按钮、链接）

- 根据页面类型定制特殊动效（如首页的英雄区、作品集的图片展示）

- 建立清晰的动效层次体系

**以用户旅程为导向**

- 首页：吸引注意力，快速传达价值

- 产品页：突出特性，促进转化

- 内容页：优化阅读体验，减少干扰

- 联系页：引导交互，提供清晰反馈

**性能与体验平衡**

- 首页可以使用更多视觉效果来创造印象

- 内容密集型页面简化动效以提升性能

- 交互型页面注重即时反馈和流畅过渡

#### JavaScript 交互

```javascript
// 特性卡片滚动动画
class FeatureAnimation {
  constructor() {
    this.cards = document.querySelectorAll('.feature-card');
    this.init();
  }

  init() {
    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach(entry => {
          if (entry.isIntersecting) {
            const delay = entry.target.dataset.delay || 0;
            setTimeout(() => {
              entry.target.classList.add('fade-in-active');
            }, delay);
            observer.unobserve(entry.target);
          }
        });
      },
      {
        threshold: 0.2,
        rootMargin: '0px 0px -50px 0px'
      }
    );

    this.cards.forEach(card => observer.observe(card));
  }
}

// 初始化
new FeatureAnimation();
```

#### JavaScript 过滤动画

```javascript
class BlogFilter {
  constructor() {
    this.filterButtons = document.querySelectorAll('.filter-btn');
    this.articles = document.querySelectorAll('.article-card');
    this.init();
  }

  init() {
    this.filterButtons.forEach(btn => {
      btn.addEventListener('click', (e) => this.handleFilter(e));
    });
  }

  handleFilter(e) {
    const filter = e.target.dataset.filter;
    
    // 更新按钮状态
    this.filterButtons.forEach(btn => btn.classList.remove('active'));
    e.target.classList.add('active');
    
    // 过滤文章
    this.articles.forEach((article, index) => {
      const category = article.dataset.category;
      const shouldShow = filter === 'all' || category === filter;
      
      if (shouldShow) {
        setTimeout(() => {
          article.style.display = 'flex';
          requestAnimationFrame(() => {
            article.classList.add('fade-in-active');
          });
        }, index * 50);
      } else {
        article.classList.remove('fade-in-active');
        setTimeout(() => {
          article.style.display = 'none';
        }, 300);
      }
    });
  }
}

// 初始化
new BlogFilter();
```

```javascript
class ReadingProgress {
  constructor() {
    this.progressBar = document.querySelector('.progress-bar');
    this.init();
  }

  init() {
    window.addEventListener('scroll', () => this.updateProgress());
    this.updateProgress();
  }

  updateProgress() {
    const windowHeight = window.innerHeight;
    const documentHeight = document.documentElement.scrollHeight;
    const scrollTop = window.pageYOffset || document.documentElement.scrollTop;
    
    const progress = (scrollTop / (documentHeight - windowHeight)) * 100;
    this.progressBar.style.width = `${Math.min(progress, 100)}%`;
  }
}
```

```javascript
class ContactForm {
  constructor() {
    this.form = document.getElementById('contactForm');
    this.submitButton = this.form.querySelector('.form-submit');
    this.successMessage = document.getElementById('successMessage');
    this.init();
  }

  init() {
    // 实时验证
    this.form.querySelectorAll('.form-input, .form-textarea').forEach(input => {
      input.addEventListener('blur', () => this.validateField(input));
      input.addEventListener('input', () => this.clearError(input));
    });

    // 字符计数
    const messageField = this.form.querySelector('#message');
    messageField.addEventListener('input', (e) => this.updateCharCount(e));

    // 表单提交
    this.form.addEventListener('submit', (e) => this.handleSubmit(e));
  }

  validateField(input) {
    const wrapper = input.closest('.input-wrapper');
    const value = input.value.trim();

    if (!value) {
      this.showError(wrapper, '此字段为必填项');
      return false;
    }

    if (input.type === 'email' && !this.isValidEmail(value)) {
      this.showError(wrapper, '请输入有效的邮箱地址');
      return false;
    }

    this.showSuccess(wrapper);
    return true;
  }

  showError(wrapper, message) {
    wrapper.classList.remove('success');
    wrapper.classList.add('error');
    const messageElement = wrapper.querySelector('.validation-message');
    messageElement.textContent = message;
  }

  showSuccess(wrapper) {
    wrapper.classList.remove('error');
    wrapper.classList.add('success');
  }

  clearError(input) {
    const wrapper = input.closest('.input-wrapper');
    wrapper.classList.remove('error');
  }

  isValidEmail(email) {
    return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
  }

  updateCharCount(e) {
    const maxLength = 500;
    const currentLength = e.target.value.length;
    const charCount = e.target.parentElement.querySelector('.char-count');
    charCount.textContent = `${currentLength} / ${maxLength}`;
    
    if (currentLength > maxLength) {
      charCount.style.color = '#f44336';
    } else {
      charCount.style.color = '#999';
    }
  }

  async handleSubmit(e) {
    e.preventDefault();

    // 验证所有字段
    const inputs = this.form.querySelectorAll('.form-input, .form-textarea');
    let isValid = true;

    inputs.forEach(input => {
      if (!this.validateField(input)) {
        isValid = false;
      }
    });

    if (!isValid) return;

    // 显示加载状态
    this.submitButton.classList.add('loading');

    try {
      // 模拟 API 调用
      await new Promise(resolve => setTimeout(resolve, 2000));

      // 隐藏表单，显示成功消息
      this.form.style.display = 'none';
      this.successMessage.classList.add('show');

      // 可选：重置表单
      setTimeout(() => {
        this.form.reset();
        this.submitButton.classList.remove('loading');
      }, 3000);

    } catch (error) {
      console.error('提交失败:', error);
      this.submitButton.classList.remove('loading');
      alert('提交失败，请稍后重试');
    }
  }
}
```

```javascript
class PortfolioLightbox {
  constructor() {
    this.lightbox = document.getElementById('lightbox');
    this.lightboxImage = this.lightbox.querySelector('.lightbox-image');
    this.currentIndex = 0;
    this.items = [];
    this.init();
  }

  init() {
    // 收集所有作品项
    this.items = Array.from(document.querySelectorAll('.portfolio-item'));

    // 绑定点击事件
    document.querySelectorAll('.view-button').forEach((btn, index) => {
      btn.addEventListener('click', (e) => {
        e.stopPropagation();
        this.open(index);
      });
    });

    // 绑定控制按钮
    this.lightbox.querySelector('.lightbox-close').addEventListener('click', () => this.close());
    this.lightbox.querySelector('.lightbox-prev').addEventListener('click', () => this.prev());
    this.lightbox.querySelector('.lightbox-next').addEventListener('click', () => this.next());

    // 点击背景关闭
    this.lightbox.addEventListener('click', (e) => {
      if (e.target === this.lightbox) {
        this.close();
      }
    });

    // 键盘导航
    document.addEventListener('keydown', (e) => {
      if (!this.lightbox.classList.contains('active')) return;

      switch(e.key) {
        case 'Escape':
          this.close();
          break;
        case 'ArrowLeft':
          this.prev();
          break;
        case 'ArrowRight':
          this.next();
          break;
      }
    });
  }

  open(index) {
    this.currentIndex = index;
    this.updateContent();
    this.lightbox.classList.add('active');
    document.body.style.overflow = 'hidden';
  }

  close() {
    this.lightbox.classList.remove('active');
    document.body.style.overflow = '';
  }

  prev() {
    this.currentIndex = (this.currentIndex - 1 + this.items.length) % this.items.length;
    this.updateContent();
  }

  next() {
    this.currentIndex = (this.currentIndex + 1) % this.items.length;
    this.updateContent();
  }

  updateContent() {
    const item = this.items[this.currentIndex];
    const img = item.querySelector('img');
    const title = item.querySelector('.portfolio-title');
    
    // 添加加载动画
    this.lightboxImage.style.opacity = '0';
    
    setTimeout(() => {
      this.lightboxImage.src = img.dataset.src || img.src;
      this.lightbox.querySelector('.lightbox-title').textContent = title.textContent;
      this.lightboxImage.style.opacity = '1';
    }, 200);
  }
}
```

```javascript
class PageTransitionManager {
  constructor() {
    this.transitionOverlay = this.createOverlay();
    this.init();
  }

  createOverlay() {
    const overlay = document.createElement('div');
    overlay.className = 'page-transition-overlay';
    overlay.innerHTML = `
      <div class="transition-logo">
        <svg><!-- 品牌 Logo --></svg>
      </div>
    `;
    document.body.appendChild(overlay);
    return overlay;
  }

  init() {
    // 拦截所有内部链接
    document.addEventListener('click', (e) => {
      const link = e.target.closest('a');
      if (link && this.isInternalLink(link)) {
        e.preventDefault();
        this.navigate(link.href);
      }
    });

    // 页面加载完成后淡入
    window.addEventListener('load', () => this.fadeIn());
  }

  isInternalLink(link) {
    return link.hostname === window.location.hostname && 
           !link.target && 
           !link.hasAttribute('download');
  }

  async navigate(url) {
    await this.fadeOut();
    window.location.href = url;
  }

  fadeOut() {
    return new Promise(resolve => {
      this.transitionOverlay.classList.add('active');
      setTimeout(resolve, 500);
    });
  }

  fadeIn() {
    setTimeout(() => {
      this.transitionOverlay.classList.remove('active');
    }, 100);
  }
}
```
